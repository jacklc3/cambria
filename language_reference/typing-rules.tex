%% ═══════════════════════════════════════════════════════════════
\section{Types}
%% ═══════════════════════════════════════════════════════════════

\[
\begin{array}{lrcl}
  \text{Value types}         & A, B   & ::= & \alpha \mid \psi \mid \void \mid \unit \mid \tpair{A}{B} \mid \teither{A}{B} \mid \tfun{A}{C} \mid \thandler{C}{D} \\[6pt]
  \text{Computation types}   & C, D   & ::= & \comp{A}{\Sig} \\[6pt]
  \text{Effect signatures}   & \Sig   & ::= & \{ \op_1 : A_1 \leadsto B_1,\; \ldots,\; \op_n : A_n \leadsto B_n \} \\[6pt]
  \text{Type variables}      & \Theta & ::= & \{ \alpha_1, \; \ldots, \; \alpha_n \} \\[6pt]
  \text{Parameter variables} & \Psi   & ::= & \{ \psi_1, \; \ldots, \; \psi_n \} \\[6pt]
  \text{Contexts}            & \G     & ::= & \emptyset \mid \G,\; x : \forall \vec{\alpha} . A
\end{array}
\]
The syntax of values and computations is as follows.
\[
\begin{array}{lrcl}
  \text{Values}       & v, w & ::= & x \mid () \mid b \mid n \mid (v_1, v_2) \mid \inl v \mid \inr v
                                     \mid \fun x \mapsto c \mid \rec f\;x \mapsto c \mid h \\[6pt]
  \text{Handler}      & h    & ::= &
    \hhandler\;\{
      \ret x_r \mapsto c_r,\;
      \op_1\; x_1\; k_1 \mapsto c_1,\;
      \ldots,\;
      \op_n\; x_n\; k_n \mapsto c_n
    \} \\[6pt]
  \text{Computations} & c, d & ::= & \ret v \mid \ddo x \leftarrow c \iin d \mid \op(v;\;x.c)
                                     \mid \with v \handle c \\[2pt]
                      &      &     & \mid\; v_1\;v_2 \mid \iif v \then c_1 \eelse c_2
                                     \mid \ccase v \oof \inl x_1 \to c_1,\;\inr x_2 \to c_2
\end{array}
\]
The form $\op(v;\;x.c)$ denotes an operation call; $\op$ is invoked with argument $v$, the result is bound to $x$, and execution continues as $c$.
The surface syntax ${!}\op\;v$ desugars to $\op(v;\;x.\ret x)$.

There are three types of contexts: one for type variables $\Theta$, one for parameter variables $\Psi$, and one for variable schemes $\G$.
We write $x : A$ for $x : \forall \vec{\alpha}. A$ if $\vec{\alpha}$ ranges over an empty set of tuples.
We assume all types appearing in a judgement conclusion are well-formed in their context.
We write $\Sig \setminus \{\op_1, \ldots, \op_n\}$ for the restriction of $\Sig$ to operations not in $\{\op_1, \ldots, \op_n\}$.

%% ═══════════════════════════════════════════════════════════════
\section{Value Typing}\label{sec:values}
%% ═══════════════════════════════════════════════════════════════

Well-formed value types:
\begin{mathpar}
  \inferrule{\alpha \in \Theta}{\tj \Theta \Psi \alpha}

  \inferrule{\psi \in \Psi}{\tj \Theta \Psi \psi}

  \inferrule{ }{\tj \Theta \Psi \tbool}

  \inferrule{ \tj \Theta \Psi A \and \tj \Theta \Psi C }{\tj \Theta \Psi \tfun{A}{C}}
\end{mathpar}
and so on. Rules for value judgements (assuming $\tj \Theta \Psi {\G, A}$).
\begin{mathpar}
  \inferrule
  {(x : \forall \vec{\alpha}.B) \in \G \\ [\tj \Theta \Psi {A_i}]_{i=1}^{|\vec{\alpha}|}}
  {\vj{\Theta}{\Psi}{\G}{x}{B[A_i / \alpha_i]_{i=1}^{|\vec{\alpha}|}}}
  \quad (\textsc{Var})

  \inferrule
    { }
    {\vj{\Theta}{\Psi}{\G}{()}{\unit}}
    \quad (\textsc{Unit})

  \inferrule
    {b \in \{\mathsf{true}, \mathsf{false}\}}
    {\vj{\Theta}{\Psi}{\G}{b}{\tbool}}
    \quad (\textsc{Bool})

  \inferrule
  {\vj{\Theta}{\Psi}{\G}{v_1}{A} \\ \vj{\Theta}{\Psi}{\G}{v_2}{B}}
  {\vj{\Theta}{\Psi}{\G}{(v_1, v_2)}{\tpair{A}{B}}}
    \quad (\textsc{Pair})

  \inferrule
  {\vj{\Theta}{\Psi}{\G}{v}{A}}
  {\vj{\Theta}{\Psi}{\G}{\inl v}{\teither{A}{B}}}
    \quad (\textsc{Inl})

  \inferrule
  {\vj{\Theta}{\Psi}{\G}{v}{B}}
  {\vj{\Theta}{\Psi}{\G}{\inr v}{\teither{A}{B}}}
    \quad (\textsc{Inr})

  \inferrule
  {\cjp{\Theta}{\Psi}{\G, x : A}{c}{C}}
    {\vj{\Theta}{\Psi}{\G}{\fun x \mapsto c}{\tfun{A}{C}}}
    \quad (\textsc{Fun})

  \inferrule
  {\cj{\Theta}{\Psi}{\G, f : \tfun{A}{\comp{B}{\Sig}},\, x : A}{c}{B}{\Sig}}
    {\vj{\Theta}{\Psi}{\G}{\rec f\; x \mapsto c}{\tfun{A}{\comp{B}{\Sig}}}}
    \quad (\textsc{Rec})

  \inferrule{
    \cj{\Theta}{\Psi}{\G, x_r : A}{c_r}{B}{\Sig'}
    \\
    \Sig \setminus \{\op_1, \ldots, \op_n\} \subseteq \Sig'
    \\\\
    \left[
      (\op_i : A_i \leadsto B_i) \in \Sig \quad
      \cj{\Theta}{\Psi}{\G,\, x_i : A_i,\, k_i : \tfun{B_i}{\comp{B}{\Sig'}}}{c_i}{B}{\Sig'}
    \right]_{i = 1}^{n}
    }
    {\vj{\Theta}{\Psi}{\G}{
      \hhandler\;\{\ret x_r \mapsto c_r,\;
        \op_i\; x_i\; k_i \mapsto c_i
      \}_{i = 1}^{n}
    }{\thandler{\comp{A}{\Sig}}{\comp{B}{\Sig'}}}}
    \quad (\textsc{Handler})
\end{mathpar}
In the handler rule, the handled operations $\{\op_1, \ldots, \op_n\}$ must appear in~$\Sig$.
Unhandled operations are forwarded, requiring $\Sig \setminus \{\op_1, \ldots, \op_n\} \subseteq \Sig'$.

%% ═══════════════════════════════════════════════════════════════
\section{Computation Typing}\label{sec:computations}
%% ═══════════════════════════════════════════════════════════════

Rules for judgements of the form $\cj{\Theta}{\Psi}{\G}{c}{A}{\Sig}$.
In \textsc{Do}, generalization quantifies the type variables $\vec{\alpha}$ that are free in $A$
but not free in the ambient context $\G$ or the shared effect signature $\Sig$.
Effect parameters $\psi$ are never generalized.

\begin{mathpar}
  \inferrule
  {\vj{\Theta}{\Psi}{\G}{v}{A}}
  {\cj{\Theta}{\Psi}{\G}{\ret v}{A}{\Sig}}
    \quad (\textsc{Return})

  \inferrule
  {\cj{\Theta}{\Psi}{\G}{c_1}{\forall \vec{\alpha}. A}{\Sig}
    \\
    \cj{\Theta}{\Psi}{\G, x : \forall\vec{\alpha}.A}{c_2}{B}{\Sig}
    }
    {\cj{\Theta}{\Psi}{\G}{\ddo x \leftarrow c_1 \iin c_2}{B}{\Sig}}
    \quad (\textsc{Do})

  \inferrule
    {(\op : \arity{A_{\op}}{B_{\op}}) \in \Sig
    \\
    \vj{\Theta}{\Psi}{\G}{v}{A_{\op}}
    \\
    \cj{\Theta}{\Psi}{\G,\, x : B_{\op}}{c}{C}{\Sig}
    }
    {\cj{\Theta}{\Psi}{\G}{\op(v;\;x.c)}{C}{\Sig}}
    \quad (\textsc{Op})

  \inferrule
  {\vj{\Theta}{\Psi}{\G}{v}{\tbool}
    \\
    \cj{\Theta}{\Psi}{\G}{c_1}{A}{\Sig}
    \\
    \cj{\Theta}{\Psi}{\G}{c_2}{A}{\Sig}
    }
    {\cj{\Theta}{\Psi}{\G}{\iif v \then c_1 \eelse c_2}{A}{\Sig}}
    \quad (\textsc{If})

  \inferrule
  {\vj{\Theta}{\Psi}{\G}{v}{\teither{A}{B}}
    \\\\
    \cjp{\Theta}{\Psi}{\G, x_1 : A}{c_1}{C}
    \\
    \cjp{\Theta}{\Psi}{\G, x_2 : B}{c_2}{C}
    }
    {\cjp{\Theta}{\Psi}{\G}{\ccase v \oof \inl x_1 \to c_1,\; \inr x_2 \to c_2}{C}}
    \quad (\textsc{Case})

  \inferrule
  {\vj{\Theta}{\Psi}{\G}{v_1}{\tfun{A}{C}}
    \\
    \vj{\Theta}{\Psi}{\G}{v_2}{A}
    }
    {\cjp{\Theta}{\Psi}{\G}{v_1\; v_2}{C}}
    \quad (\textsc{App})

  \inferrule{
    \vj{\Theta}{\Psi}{\G}{v}{\thandler{C[\vec{T} / \vec{\psi}]}{D}}
    \\
    \cjp{\Theta}{\Psi, \vec{\psi}}{\G}{c}{C}
    \\
    \left[\tj \Theta \Psi {T_j}\right]_{j=1}^{|\vec{T}|}
  }{
    \cjp{\Theta}{\Psi}{\G}{\with v, \{\vec{\psi}\mapsto\vec{T}\} \handle c}{D}
  }
  \quad (\textsc{Handle})

  \inferrule{
    \cj {\Theta, \vec{\alpha}} \Psi \G c A \Sig
  }{
    \cj \Theta \Psi \G c {\forall \vec{\alpha}.A} \Sig
  }
  \quad (\textsc{Gen})
\end{mathpar}


In \textsc{Handle}, the body $c$ is typed in the extended parameter context $\Psi, \vec{\psi}$.
Inside the body, the $\psi_j$ are unknown types. They are only resolved by a handle clause.
A handle clause these types with parameter instantiations $\vec{\psi} \mapsto \vec{T}$,
which are collected into a substitution $\sigma = \{\vec{\psi} \mapsto \vec{T}\}$.
The handler domain $C[\vec{T} / \vec{\psi}]$ states that applying the parameter substitution to the body's type yields the handler's input type.
Since the output type $\comp{B}{\Sig'}$ is typed in the outer $\G$ (which does not contain $\vec{\psi}$),
well-formedness of the conclusion directly prevents the instantiated parameters from escaping.
When $\vec{\psi} = \emptyset$, the rule reduces to the standard handler application rule.

%% ═══════════════════════════════════════════════════════════════
\section{Operational Semantics}\label{sec:opsem}
%% ═══════════════════════════════════════════════════════════════

Evaluation contexts identify where the next reduction step occurs:
\[
  E \;::=\; \square
    \mid \ddo x \leftarrow E \iin c
    \mid \with v \handle E
\]
Single-step reduction $c \leadsto c'$ is defined by the axioms below, closed under the congruence rule: if $c \leadsto c'$ then $E[c] \leadsto E[c']$.

\paragraph{Pure reductions.}
\[
\begin{array}{lcll}
  (\fun x \mapsto c)\;v
    & \leadsto & c[v/x]
    & (\beta\text{-fun}) \\[4pt]
  (\rec f\;x \mapsto c)\;v
    & \leadsto & c\bigl[v/x,\;(\rec f\;x \mapsto c)/f\bigr]
    & (\beta\text{-rec}) \\[4pt]
  \ddo x \leftarrow \ret v \iin c
    & \leadsto & c[v/x]
    & (\beta\text{-do-ret}) \\[4pt]
  \ddo x \leftarrow \op(v; y.c) \iin c'
    & \leadsto & \op(v; y.\ddo x \leftarrow c \iin c')
    & (\beta\text{-do-op}) \\[4pt]
  \iif \mathsf{true} \then c_1 \eelse c_2
    & \leadsto & c_1
    & (\beta\text{-if}_T) \\[4pt]
  \iif \mathsf{false} \then c_1 \eelse c_2
    & \leadsto & c_2
    & (\beta\text{-if}_F) \\[4pt]
  \ccase \inl v \oof \inl x_1 \to c_1,\;\inr x_2 \to c_2
    & \leadsto & c_1[v/x_1]
    & (\beta\text{-inl}) \\[4pt]
  \ccase \inr v \oof \inl x_1 \to c_1,\;\inr x_2 \to c_2
    & \leadsto & c_2[v/x_2]
    & (\beta\text{-inr})
\end{array}
\]

\paragraph{Handler reductions.}

Let $h = \hhandler\;\{\ret x_r \mapsto c_r,\;\op_i\;x_i\;k_i \mapsto c_i\}_{i=1}^{n}$.

\[
\begin{array}{lcll}
  \with h, \sigma \handle \ret v
    & \leadsto & c_r[v/x_r]
    & (\beta\text{-ret}) \\[4pt]
    \with h,\sigma \handle \op_i(v;\;y.c)
    & \leadsto & c_i\bigl[v/x_i,\;(\fun y \mapsto \with h \handle c)/k_i\bigr]
    & (\beta\text{-op}) \\[4pt]
  \with h,\sigma \handle \op(v;\;y.c)
    & \leadsto & \op(v;\;y.\;\with h \handle c)
    & (\beta\text{-fwd})
\end{array}
\]

\noindent
($\beta$-op) applies when $\op \in \{\op_1, \ldots, \op_n\}$, while
($\beta$-fwd) applies when $\op \notin \{\op_1, \ldots, \op_n\}$.
The continuation supplied to the clause in ($\beta$-op) is $\fun y \mapsto \with h \handle c$,
reinstating the handler to give \emph{deep} handler semantics.

\section{Metatheory}\label{sec:meta}
%% ═══════════════════════════════════════════════════════════════

\begin{definition}
  \emph{Parameter erasure} is a map $\lfloor - \rfloor$ on values and computations given by
  \[
    \bigl\lfloor \with h, \sigma \handle c \bigr\rfloor
    \;=\;
    \with \lfloor h \rfloor, \emptyset \handle \lfloor c \rfloor
  \]
  and structural identity on all other value and computation forms.
\end{definition}


\begin{lemma}[Parameter Erasure]\label{lem:erasure}
  For all computations $c$ and $c'$,\; $c \leadsto c'$ if and only if $\lfloor c \rfloor \leadsto \lfloor c' \rfloor$.
\end{lemma}

\begin{proof}
  The parameter instantiations $\sigma$ in a \textsc{Handle} term do not appear in
  the conclusion of any reduction rule. Consequently, erasing all parameter annotations
  from a well-typed closed program yields a program with an identical reduction sequence.
\end{proof}

\begin{lemma}[Parameter Substitution]\label{lem:param-subst}
  If $\tj{\Theta}{\Psi}{T}$, then:
  \begin{enumerate}
    \item If $\vj{\Theta}{\Psi, \psi}{\G}{v}{A}$ then $\vj{\Theta}{\Psi}{\G[T/\psi]}{v}{A[T/\psi]}$.
    \item If $\cj{\Theta}{\Psi, \psi}{\G}{c}{A}{\Sig}$ then $\cj{\Theta}{\Psi}{\G[T/\psi]}{c}{A[T/\psi]}{\Sig[T/\psi]}$.
  \end{enumerate}
\end{lemma}

\begin{proof}
  By induction on the typing derivation.
  The base case $\psi \in \Psi, \psi$ is immediate from the hypothesis $\tj{\Theta}{\Psi}{T}$.
  All other cases follow by straightforward induction: $\psi$ is a type variable and $[T/\psi]$ distributes uniformly over all type constructors without affecting the term structure.
\end{proof}

The following results extend~\cite{kammar_handlers_2013} to the parameterized setting.
Progress and preservation for all rules except \textsc{Handle} carry over unchanged;
the ($\beta$-op) case of preservation uses \Cref{lem:param-subst}.

\begin{theorem}[Progress]\label{thm:progress}
  If $\cj{\emptyset}{\emptyset}{\emptyset}{c}{A}{\emptyset}$, then either $c = \ret v$ for some value $v$,
  or there exists $c'$ such that $c \leadsto c'$.
\end{theorem}

\begin{proof}
  By \Cref{lem:erasure}, $\lfloor c \rfloor$ is a well-typed program in the plain algebraic effects system with an identical reduction sequence.
  Progress for $\lfloor c \rfloor$ follows from \cite{kammar_handlers_2013}: either $\lfloor c \rfloor = \ret v$, in which case $c = \ret v'$ for some value $v'$,
  since erasure does not alter the term structure; or $\lfloor c \rfloor \leadsto d$, in which case \Cref{lem:erasure} gives $c \leadsto c'$ for some $c'$.
\end{proof}

\begin{theorem}[Preservation]\label{thm:preservation}
  If $\cjp{\Theta}{\Psi}{\G}{c}{C}$ and $c \leadsto c'$, then $\cjp{\Theta}{\Psi}{\G}{c'}{C}$.
\end{theorem}

\begin{proof}
  By induction on the derivation of $c \leadsto c'$.
  All cases except ($\beta$-op) are identical to \cite{kammar_handlers_2013}.
  For ($\beta$-op), the premise of the reduction rule arises in the following way:
  \[
    {\scriptsize
    \inferrule{
      \vj{\Theta}{\Psi}{\G}{h}{\thandler{\comp{A}{\Sig}}{\comp{B}{\Sig'}}}
      \\
      \inferrule{
        (\op : \widetilde{A}_\op \to \widetilde{B}_\op) \in \widetilde \Sig
        \\
        \vj{\Theta}{\Psi, \dom(\sigma)}{\G}{v}{\widetilde{A}_\op}
        \\
        \cj{\Theta}{\Psi, \dom(\sigma)}{\G, y : \widetilde{B}_\op}{c}{\widetilde{A}}{\widetilde \Sig}
      }{
        \cj{\Theta}{\Psi, \dom(\sigma)}{\G}{\op(v; y.c)}{\widetilde A}{\widetilde \Sig}
      }
    }{
      \cj{\Theta}{\Psi}{\G}{\with h, \sigma \handle \op(v; y.c)}{B}{\Sig'}
    }}
  \]
  where $\sigma(\comp{\widetilde A}{\widetilde \Sig}) = \comp{A}{\Sig}$.
  Applying \Cref{lem:param-subst} for each $\psi \in \dom(\sigma)$ yields the following typing derivation.
  \[
    {\scriptsize
    \inferrule{
      \vj{\Theta}{\Psi}{\G}{h}{\thandler{\comp{A}{\Sig}}{\comp{B}{\Sig'}}}
      \\
      \inferrule{
        (\op : A_\op \to B_\op) \in \Sig
        \\
        \vj{\Theta}{\Psi}{\G}{v}{A_\op}
        \\
        \cj{\Theta}{\Psi}{\G, y : B_\op}{c}{A}{\Sig}
      }{
        \cj{\Theta}{\Psi}{\G}{\op(v; y.c)}{A}{\Sig}
      }
    }{
      \cj{\Theta}{\Psi}{\G}{\with h, \emptyset \handle \op(v; y.c)}{B}{\Sig'}
    } }
  \]
  So we can appeal to~\cite{kammar_no_2017} for preservation of this reduction rule.
\end{proof}

\section{Type Inference}\label{sec:inference}
%% ═══════════════════════════════════════════════════════════════

The inference algorithm extends Hindley--Milner~\cite{damas_principal_1982} allowing for explicit effect annotations.
Otherwise, the algorithm is ordinary HM inference.
Two completeness results relate the algorithm to the declarative system of \Cref{sec:computations}.

\begin{lemma}[Ground Completeness] \label{thm:ground-complete}
  Suppose every handle expression in $t$ has $\sigma = \emptyset$.
  If\/ $\cjp{\Theta}{\Psi}{\G}{t}{C}$ is derivable declaratively, the algorithm infers
  a type $C'$ of which $C$ is an instance.
\end{lemma}

\begin{proof}
  Any derivation in this fragment is identical to a derivation in the plain algebraic effects system,
  for which completeness has been established~\cite{kammar_no_2017}.
\end{proof}

The second result covers the full parameterized system.
An \emph{annotated term} is one in which every handle expression
$\with v \handle c$ carries an explicit type annotation
$\with v \handle (c : C)$.

\begin{theorem}[Parameterized Completeness]\label{thm:param-complete}
  Let $t$ be an annotated term and let $\G$ contain principal types for all variables.
  If $\cjp{\Theta}{\Psi}{\G}{t}{C}$ is derivable in the declarative system, then the
  algorithm infers a principal type $C'$ of which $C$ is an instance.
\end{theorem}

\begin{proof}
  By induction on the size of the declarative derivation. The only non-standard cases are:
  \begin{itemize}
    \item \emph{\textsc{Do}.}
          By \emph{gen admissibility}~\cite{damas_principal_1982}, any declarative derivation
          can be rearranged so that \textsc{Gen} appears only immediately below
          \textsc{Do}. This extends to our system because effect parameters are never generalized.

    \item \emph{\textsc{Handle}.}
          The algorithm infers the handler type
          $\sigma(C) \xRightarrow{\sigma} D$ by the induction hypothesis, and checks the body against the explicitly given $C$
          where the parameter context is extended by $\dom(\sigma)$.
          Verification that $\sigma(C)$ is the handler's input type is a single pass of
          substitution followed by equality checking.
          No additional unification is required.
  \end{itemize}
\end{proof}

The annotation in \Cref{thm:param-complete} specifies the full type $C$ of the body.
The following theorem strengthens this to a minimal annotation consisting only of the
operation arities that mention $\dom(\sigma)$.

\begin{definition}[$\sigma$-relevant operations] \label{def:relevant}
  Given a handler $v$ with parameter substitution $\sigma$, the
  \emph{$\sigma$-relevant subset} of the handled signature is
  \[
    \ops_\sigma(v)
    \;=\;
    \bigl\{\,\op_i : A_i \leadsto B_i \mid
      \op_i \text{ handled by } v
      \text{ and }
      \dom(\sigma) \cap (\fpv(A_i) \cup \fpv(B_i)) \neq \emptyset
    \,\bigr\}.
  \]
  A \emph{minimally annotated term} is one in which each handle expression
  $\with v \handle c$ carries an annotation listing only $\ops_\sigma(v)$;
  the return type and all other operations are omitted.
\end{definition}

\begin{corollary}[Minimal Annotation Completeness]\label{thm:minimal-complete}
  Let $t$ be a minimally annotated term and let $\G$ contain principal types for all
  variables.
  If\/ $\cjp{\Theta}{\Psi}{\G}{t}{A}$ is derivable declaratively, then the algorithm
  infers a principal type $A'$ of which $A$ is an instance.
\end{corollary}

\begin{proof}
  We extend the proof of \Cref{thm:param-complete}, strengthening the \textsc{Handle} case.
  Partition the body's effect signature into three groups:

  \begin{enumerate}
    \item \emph{$\sigma$-relevant handled ops.}
          The annotation seeds each $\psi_j \in \dom(\sigma)$ as a rigid constant
          into $\Psi$, and the argument proceeds as in \Cref{thm:param-complete}.

    \item \emph{$\sigma$-free handled ops.}
          For any such $\op_i : A_i \leadsto B_i$ we have
          $\dom(\sigma) \cap (\fpv(A_i) \cup \fpv(B_i)) = \emptyset$,
          so $\sigma(A_i) = A_i$ and $\sigma(B_i) = B_i$.
          Typing subterms against the handler's concrete arity already gives the correct
          abstract type; no annotation is needed.
          Moreover, since $\psi$ does not appear in $A_i$ or $B_i$, every unification
          constraint arising from $\op_i$ is $\sigma$-free, so the rigid $\psi$
          established by group~(1) is not disturbed.

    \item \emph{Forwarded ops.}
          Forwarded operations flow from the body's signature into $D$ unchanged, so
          their arities are $\sigma$-free.
          The argument is then identical to group~(2).
  \end{enumerate}
  \emph{Return type.}
  With each $\psi_j$ rigid in $\Psi$, the unification variable the algorithm introduces
  for the return type is constrained only by value-level subterms of $c$.
  Since $\psi_j$ acts as a ground constant to the unifier, the resulting value type is
  principal without any annotation on the return position.
\end{proof}

The annotation $\ops_\sigma(v)$ is a term-independent sufficient condition.
However, individual operations in $\ops_\sigma(v)$ may be omitted when
$\psi$ propagates to their arity via term-level value flow.
For example, consider operations $\mathsf{get} : \unit \leadsto \psi$ and
$\mathsf{set} : \psi \leadsto \unit$. Annotating only $\mathsf{get}$ in a computation body
$\ddo x \leftarrow {!}\mathsf{get}\; () \iin {!}\mathsf{set}\;x$ permits the algorithm
to infer $\mathsf{set} : \psi \leadsto \unit$ from $x : \psi$.

For this reason, Cambria allows individual effect operations to be types with the
syntax,
\[
\effect !\op : A \leadsto B. \; c,
\]
but enforces no constraints on which operations need to be annotated,
leaving the programmer to manage the interface on their own.
